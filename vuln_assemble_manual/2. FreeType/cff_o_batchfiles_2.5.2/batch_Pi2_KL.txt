#! /bin/bash

rm -rf /home/pi/outputBin
mkdir /home/pi/outputBin
cd /home/pi/

tar xvzf freetype-2.5.2.tar.gz
mkdir freetype

cd /home/pi/freetype-2.5.2
./configure --prefix=/home/pi/freetype --without-png
find ./ -name "*" -exec sed -i 's/O2/O0/g' {} \;
make
make install
cd objs
